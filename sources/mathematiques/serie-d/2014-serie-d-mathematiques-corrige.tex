\exo{1}{}
\textbf{1.} Soit $z = x + iy$. Le conjugué de $z$ est $\overline{z} = x - iy$.

\textbf{2. a.} Posons $Z = r\,e^{i\theta}$.
\[ Z^3 = 1 \Longleftrightarrow \begin{cases} r = 1 \\ \theta_k = \dfrac{2k\pi}{3},\ k \in \Z \end{cases} \]
D'où $z_0 = 1$ ; \quad $z_1 = e^{i\frac{2\pi}{3}} = -\dfrac{1}{2} + i\dfrac{\sqrt{3}}{2}$ ; \quad $z_2 = e^{i\frac{4\pi}{3}} = -\dfrac{1}{2} - i\dfrac{\sqrt{3}}{2}$.

\textbf{b.} Les solutions non réelles sont conjuguées : $\overline{z_1} = -\dfrac{1}{2} - i\dfrac{\sqrt{3}}{2} = z_2$.

\textbf{3.} On remarque que $Z_1 = 2z_2$, donc $Z_1^3 = 2^3 z_2^3$. Or $z_2^3 = 1$, d'où $Z_1^3 = 8$.

\textbf{4. a.} Si $z$ est solution de $(E)$, alors $z^3 = 1$, donc $(2z)^3 = 8$ : $2z$ est solution de $(E')$. Les solutions de $(E')$ sont les doubles de celles de $(E)$ : $z'_0 = 2$ ; $z'_1 = -1 + i\sqrt{3}$ ; $z'_2 = -1 - i\sqrt{3}$.

\textbf{b.} Comme $z_1'^3 = z_2'^3 = 8$, alors $\left(\dfrac{z'_1}{z'_2}\right)^3 = \dfrac{8}{8} = 1$ : $\dfrac{z'_1}{z'_2}$ est solution de $(E)$.

\exo{2}{}
\textbf{1.} $f(\vi)$ a pour coordonnées $(0, 1, 1)$, donc $f(\vi) = \vj + \vec{k}$. De même $f(\vj) = \vi + \vj$ et $f(\vec{k}) = \vi + \vj$.

\textbf{2.} Matrice de $f$ dans la base $(\vi, \vj, \vec{k})$ :
\[ M = \begin{pmatrix} 0 & 1 & 1 \\ 1 & 1 & 1 \\ 1 & 0 & 0 \end{pmatrix} \]

\textbf{3. a.} Un sous-ensemble $\mathscr{E}$ de $\R^3$ est un sous-espace vectoriel si : (i) $\mathscr{E} \ne \emptyset$ ; (ii) $\forall \vec{u}, \vec{v} \in \mathscr{E}$, $\vec{u} + \vec{v} \in \mathscr{E}$ ; (iii) $\forall \vec{u} \in \mathscr{E}$, $\forall \lambda \in \R$, $\lambda\vec{u} \in \mathscr{E}$.

\textbf{b.} (i) $(0,0,0) \in \mathscr{H}$ car $0 - 0 + 0 = 0$, donc $\mathscr{H} \ne \emptyset$. (ii) Si $\vec{u}(x,y,z)$ et $\vec{v}(x',y',z') \in \mathscr{H}$ : $(x+x') - (y+y') + (z+z') = (x-y+z) + (x'-y'+z') = 0$, donc $\vec{u}+\vec{v} \in \mathscr{H}$. (iii) $\lambda x - \lambda y + \lambda z = \lambda(x-y+z) = 0$, donc $\lambda\vec{u} \in \mathscr{H}$.

\textbf{4.} $(x,y,z)$ appartient au noyau de $f$ ssi $f(x,y,z) = \vec{0}$ :
\[ \begin{cases} y + z = 0 \\ x + y + z = 0 \\ x = 0 \end{cases} \Longleftrightarrow \begin{cases} z = -y \\ x = 0 \end{cases} \]
Un vecteur du noyau s'écrit $(0, y, -y) = y(0, 1, -1)$. Le noyau est la droite vectorielle engendrée par $\vec{e_1} = (0, 1, -1)$.

\textbf{5.} $(x',y',z')$ est dans l'image ssi il existe $(x,y,z)$ avec $y+z=x'$, $x+y+z=y'$, $x=z'$. On relève $x' - y' + z' = 0$, soit $y' = x' + z'$. D'où $(x',y',z') = x'(1,1,0) + z'(0,1,1)$ : l'image de $f$ est le plan engendré par $\vec{e_2} = (1,1,0)$ et $\vec{e_3} = (0,1,1)$, d'équation $x - y + z = 0$.

\exo{3}{}
\textbf{1.} Comme $e^x + 1 \ne 0$, la fonction $x \mapsto \dfrac{e^x}{e^x + 1} - x$ existe sur $\R$. Donc $E_g = \R$.

\textbf{2.} $g$ est dérivable sur $\R$ et $g'(x) = \dfrac{e^x(e^x+1) - e^x\cdot e^x}{(e^x+1)^2} - 1 = -\dfrac{e^{2x} + e^x + 1}{(e^x+1)^2} < 0$.

\textbf{3.} Avec $\lim\limits_{x\to-\infty} g = +\infty$ et $\lim\limits_{x\to+\infty} g = -\infty$ :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline $x$ & $-\infty$ & & $\alpha$ & & $+\infty$ \\\hline $g'(x)$ & & $-$ & & $-$ & \\\hline $g$ & $+\infty$ & $\searrow$ & $0$ & $\searrow$ & $-\infty$ \\\hline
\end{tabular}
\end{center}

\textbf{4.} $g$ est continue, strictement décroissante sur $\R$, et $0 \in ]-\infty;+\infty[$. D'après le théorème des valeurs intermédiaires, il existe un unique $\alpha$ tel que $g(\alpha) = 0$, soit $\dfrac{e^\alpha}{e^\alpha + 1} = \alpha$ : $\alpha$ est solution de $\dfrac{e^x}{e^x+1} = x$.

\textbf{5. a.} $h(x) = \dfrac{e^x}{e^x+1}$, $h'(x) = \dfrac{e^x}{(e^x+1)^2} > 0$ pour tout $x \in \R$.

\textbf{b.} Avec $\lim\limits_{x\to-\infty} h = 0$ et $\lim\limits_{x\to+\infty} h = 1$ :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccc|}
\hline $x$ & $-\infty$ & & $+\infty$ \\\hline $h'(x)$ & & $+$ & \\\hline $h$ & $0$ & $\nearrow$ & $1$ \\\hline
\end{tabular}
\end{center}

\textbf{c.} $h$ est strictement croissante sur $\R$ avec $\lim\limits_{-\infty} h = 0$, $\lim\limits_{+\infty} h = 1$. Donc $\forall x \in \R$, $0 \le h(x) \le 1$.

\textbf{6. a.} \underline{$\forall n$, $u_n \le 1$ (récurrence) :} $u_0 = 0 \le 1$ et $u_1 = h(u_0) = \tfrac{1}{2} \le 1$. Si $u_n \le 1$, alors $u_{n+1} = h(u_n) \le 1$ (d'après 5.c.). Donc $u_n \le 1$ pour tout $n$.

\textbf{b.} \underline{$(u_n)$ croissante :} $u_0 = 0 \le u_1 = \tfrac{1}{2}$. Si $u_n \le u_{n+1}$, comme $h$ est croissante, $h(u_n) \le h(u_{n+1})$, soit $u_{n+1} \le u_{n+2}$. Donc $(u_n)$ est croissante.

\textbf{c.} $(u_n)$ est croissante et majorée par 1, donc convergente.

\underline{$\lim u_n = \alpha$ :} (i) Pour $x \in [0;1]$ : $e^x \ge 1$, $(e^x+1)^2 \ge 4$, et $e^x \le e$, d'où $0 < h'(x) = \dfrac{e^x}{(e^x+1)^2} \le \dfrac{e}{4}$. (ii) Par l'inégalité des accroissements finis, $|h(x) - h(y)| \le \dfrac{e}{4}|x - y|$ sur $[0;1]$ ; comme $h(\alpha) = \alpha$ et $u_n, \alpha \in [0;1]$, $|u_{n+1} - \alpha| \le \dfrac{e}{4}|u_n - \alpha|$. (iii) Par récurrence, $|u_n - \alpha| \le \left(\dfrac{e}{4}\right)^n|u_0 - \alpha|$. Comme $\dfrac{e}{4} < 1$, par encadrement $\lim\limits_{n\to+\infty}|u_n - \alpha| = 0$, donc $\lim\limits_{n\to+\infty} u_n = \alpha$.

\exo{4}{}
On note $(x_i, n_{i\bullet})$ et $(y_j, n_{\bullet j})$ les distributions marginales, avec $N = \sum_i n_{i\bullet} = \sum_j n_{\bullet j}$.

\textbf{1.} Séries marginales :
\begin{center}\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|c|c|}\hline $x$ & $-1$ & $0$ & $2$ \\\hline $n_{i\bullet}$ & 3 & $a$ & 2 \\\hline\end{tabular}\qquad
\begin{tabular}{|c|c|c|}\hline $y$ & $1$ & $3$ \\\hline $n_{\bullet j}$ & 3 & $a+2$ \\\hline\end{tabular}
\end{center}

\textbf{2.} $\overline{x} = \dfrac{3(-1) + a(0) + 2(2)}{a + 5} = \dfrac{1}{5 + a}$ ; $\overline{y} = \dfrac{3(1) + (a+2)(3)}{5 + a} = \dfrac{3a + 9}{a + 5}$. On cherche $a$ tel que $\overline{x} = \dfrac{1}{6}$ et $\overline{y} = 2$ : d'où $a = 1$.

\textbf{3.} Pour $a = 1$ : $\overline{x} = \dfrac{1}{6}$, $\overline{y} = 2$.
\[ V(x) = \frac{3(-1)^2 + 1(0)^2 + 2(2)^2}{6} - \left(\frac{1}{6}\right)^2 = \frac{65}{36} \;;\quad V(y) = \frac{3(1)^2 + 3(3)^2}{6} - 2^2 = 1 \]
\[ \mathrm{Cov}(x,y) = \frac{-1 - 6 + 0 + 0 + 4 + 0}{6} - \frac{1}{6}\times 2 = -\frac{5}{6} \]
Le coefficient de corrélation linéaire est $\rho_{x,y} = \dfrac{\mathrm{Cov}(x,y)}{\sqrt{V(x)}\sqrt{V(y)}} = \dfrac{-5/6}{\sqrt{65/36}} = -\dfrac{5}{\sqrt{65}} \approx -0{,}62$.
