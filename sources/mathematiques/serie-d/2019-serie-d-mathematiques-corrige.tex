\exo{1}{}
\textbf{1. a.} $Z^2 - 4Z + 8 = 0$ a pour discriminant réduit $\Delta' = (-2)^2 - 8 = -4 = (2i)^2$. Racines : $Z_1 = 2 - 2i$ et $Z_2 = 2 + 2i$.

\textbf{b.} $|Z_1| = 2\sqrt{2}$, d'où $Z_1 = 2\sqrt{2}\left(\cos\left(-\tfrac{\pi}{4}\right) + i\sin\left(-\tfrac{\pi}{4}\right)\right)$.

\textbf{2. a.} $U = \dfrac{Z_B}{Z_A} = \dfrac{2 + 2i}{2 - 2i} = \dfrac{1+i}{1-i} = i$.

\textbf{b.} $\left|\dfrac{Z_B - Z_O}{Z_A - Z_O}\right| = |i| = 1$, donc $OA = OB$. $(\vv{OA}, \vv{OB}) \equiv \arg(i) = \dfrac{\pi}{2}\ [2\pi]$. Le triangle $OAB$ est rectangle isocèle en $O$.

\textbf{3. a.} $Z' = e^{i\frac{\pi}{3}}Z$ avec $Z_O = 0$ : $f$ est la rotation de centre $O$ et d'angle $\dfrac{\pi}{3}$.

\textbf{b.} $Z_A = 2\sqrt{2}\,e^{-i\frac{\pi}{4}}$, d'où $Z_{A'} = e^{i\frac{\pi}{3}}Z_A = 2\sqrt{2}\,e^{i\frac{\pi}{12}} = 2\sqrt{2}\left(\cos\tfrac{\pi}{12} + i\sin\tfrac{\pi}{12}\right)$. Forme algébrique : $Z_{A'} = \left(\tfrac{1}{2} + i\tfrac{\sqrt{3}}{2}\right)(2 - 2i) = (1 + \sqrt{3}) + i(-1 + \sqrt{3})$.

\textbf{c.} Par identification : $\cos\dfrac{\pi}{12} = \dfrac{1 + \sqrt{3}}{2\sqrt{2}} = \dfrac{\sqrt{2} + \sqrt{6}}{4}$ et $\sin\dfrac{\pi}{12} = \dfrac{-1 + \sqrt{3}}{2\sqrt{2}} = \dfrac{-\sqrt{2} + \sqrt{6}}{4}$.

\exo{2}{}
\textbf{1.} $f(\vi) = 2\vi - \vj$ et $f(\vj) = 3\vi - 2\vj$.

\textbf{2.} Matrice de $f$ dans $(\vi,\vj)$ : $\begin{pmatrix} 2 & 3 \\ -1 & -2 \end{pmatrix}$.

\textbf{3.} $f(\vec{V})$ pour $\vec{V}(3, -4)$ : $x' = 2(3) + 3(-4) = -6$, $y' = -3 - 2(-4) = 5$, soit $\vec{V'}(-6, 5)$.

\textbf{4.} $\begin{vmatrix} 2 & 3 \\ -1 & -2 \end{vmatrix} = -1 \ne 0$ : $f$ est bijectif (automorphisme).

\textbf{5. a.} $f\circ f(\vi) = \vi$ et $f\circ f(\vj) = \vj$.

\textbf{b.} $(\vi,\vj)$ est une base et $f\circ f = \mathrm{id}$ : $f$ est une symétrie vectorielle.

\textbf{c.} \underline{Base :} $f(\vec{u}) = \vec{u} \Leftrightarrow x + 3y = 0$. \underline{Direction :} $f(\vec{u}) = -\vec{u} \Leftrightarrow x + y = 0$.

\partieT{II}
\textbf{1. a.} $f(x) = g(x) \Leftrightarrow \dfrac{\ln x}{x} = 0 \Leftrightarrow x = 1$.

\textbf{b.} Sur $]0;1[$, $f(x) - g(x) = \dfrac{\ln x}{x} < 0$ : $(\mathcal{C})$ est en dessous de $(\mathcal{C}')$. Sur $]1;+\infty[$, $\dfrac{\ln x}{x} > 0$ : $(\mathcal{C})$ est au-dessus de $(\mathcal{C}')$.

\textbf{2. a.} $\lim\limits_{x\to 0^+} f(x) = -\infty$ ; $\lim\limits_{x\to+\infty} f(x) = 0$.

\textbf{b.} $\forall x \in \R_+^*$, $f'(x) = \dfrac{\frac{1}{x}\cdot x - \ln x}{x^2} - \dfrac{1}{x^2} = -\dfrac{\ln x}{x^2}$, du signe de $-\ln x$ (s'annule en $x = 1$).
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline $x$ & $0$ & & $1$ & & $+\infty$ \\\hline $f'(x)$ & $\|$ & $+$ & $0$ & $-$ & \\\hline $f$ & $-\infty$ & $\nearrow$ & $1$ & $\searrow$ & $0$ \\\hline
\end{tabular}
\end{center}

\partieT{III}
\textbf{1.} $\forall x \in \R_+^*$, $h'(x) = \dfrac{1}{2}\cdot 2\cdot\dfrac{1}{x}\cdot\ln x = \dfrac{\ln x}{x} = k(x)$ : $h$ est une primitive de $k$ sur $\R_+^*$.

\textbf{2.} $\displaystyle\int_1^e (f(x) - g(x))\,dx = \int_1^e h'(x)\,dx = [h(x)]_1^e = \dfrac{1}{2}\ \text{u.a} = 2\ \text{cm}^2$.
