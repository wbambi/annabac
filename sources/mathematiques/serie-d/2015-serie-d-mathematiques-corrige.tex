\exo{1}{}
\textbf{1.} Les racines carrées de $u = 6 + 6i\sqrt{3}$ sont les $z = x + iy$ tels que $z^2 = u$, soit $x^2 - y^2 = 6$, $xy = 3\sqrt{3}$, et $x^2 + y^2 = |u| = 12$. On obtient :
\[ \begin{cases} x^2 - y^2 = 6 \\ x^2 + y^2 = 12 \\ xy = 3\sqrt{3} \end{cases} \Rightarrow x = \pm 3,\ y = \pm\sqrt{3}\ (\text{même signe}) \]
D'où les racines $\alpha_1 = 3 + i\sqrt{3}$ et $\alpha_2 = -3 - i\sqrt{3}$.

\textbf{2. a.} En remplaçant $Z = -\dfrac{1}{2}$ dans $4Z^3 - 6i\sqrt{3}Z^2 - (9 + 3i\sqrt{3})Z - 4$ :
\[ \frac{-4 - 12i\sqrt{3} + 36 + 12i\sqrt{3} - 32}{8} = 0. \]

\textbf{b.} On cherche $a, b, c$ tels que $4Z^3 - 6i\sqrt{3}Z^2 - (9 + 3i\sqrt{3})Z - 4 = \left(Z + \tfrac{1}{2}\right)(aZ^2 + bZ + c)$. Par identification : $a = 4$, $c = -8$, $b = -2 - 6i\sqrt{3}$. D'où le facteur $4Z^2 - 2(1 + 3i\sqrt{3})Z - 8$.

\underline{Résolution de $4Z^2 - 2(1 + 3i\sqrt{3})Z - 8 = 0$ :} le discriminant réduit est $\Delta' = (1 + 3i\sqrt{3})^2 + 32 = 6 + 6i\sqrt{3} = u = (3 + i\sqrt{3})^2$. D'où
\[ Z_2 = \frac{1 + 3i\sqrt{3} - 3 - i\sqrt{3}}{4} = -\frac{1}{2} + i\frac{\sqrt{3}}{2},\quad Z_3 = \frac{1 + 3i\sqrt{3} + 3 + i\sqrt{3}}{4} = 1 + i\sqrt{3}. \]
Les solutions de $(E)$ sont $Z_1 = -\dfrac{1}{2}$, $Z_2 = -\dfrac{1}{2} + i\dfrac{\sqrt{3}}{2}$, $Z_3 = 1 + i\sqrt{3}$.

\textbf{3. a.} Soit $Z' = aZ + b$ l'expression de $S$.
\[ \begin{cases} S(A) = B \\ S(B) = C \end{cases} \Rightarrow a = \frac{Z_C - Z_B}{Z_B - Z_A} = \frac{\frac{3}{2} + i\frac{\sqrt{3}}{2}}{i\frac{\sqrt{3}}{2}} = 1 - i\sqrt{3},\quad b = Z_B - aZ_A = 0. \]
D'où $Z' = (1 - i\sqrt{3})Z$.

\textbf{b.} $1 - i\sqrt{3} = 2e^{-i\frac{\pi}{3}}$ : $S$ est la similitude directe de centre $O$ (affixe 0), de rapport 2 et d'angle $-\dfrac{\pi}{3}$.

\exo{2}{}
\textbf{1.} $\det_{(\vi,\vj)}(\vec{e_1}, \vec{e_2}) = \begin{vmatrix} 1 & 2 \\ 1 & 3 \end{vmatrix} = 1 \ne 0$, donc $\{\vec{e_1}, \vec{e_2}\}$ est une base de $\mathscr{E}$.

\textbf{2.} Matrice de $f$ dans la base $(\vec{e_1}, \vec{e_2})$ : $\begin{pmatrix} 1 & 0 \\ 0 & 0 \end{pmatrix}$.

\textbf{3. a.} De $\begin{cases} f(\vec{e_1}) = \vec{e_1} \\ f(\vec{e_2}) = \vec{0} \end{cases}$, c'est-à-dire $\begin{cases} f(\vi) + f(\vj) = \vi + \vj \\ 2f(\vi) + 3f(\vj) = \vec{0} \end{cases}$, on tire $f(\vi) = 3\vi + 3\vj$ et $f(\vj) = -2\vi - 2\vj$.

\textbf{b.} $f(\vec{u}) = f(x\vi + y\vj) = x(3\vi + 3\vj) + y(-2\vi - 2\vj) = (3x - 2y)\vi + (3x - 2y)\vj$. D'où $\begin{cases} x' = 3x - 2y \\ y' = 3x - 2y \end{cases}$.

\textbf{c.} $f\circ f(\vec{u}) = (3x - 2y)f(\vi) + (3x - 2y)f(\vj) = (3x - 2y)\vi + (3x - 2y)\vj = f(\vec{u})$.

\textbf{d.} $f$ est un endomorphisme vérifiant $f \circ f = f$ : c'est une projection vectorielle, de base l'ensemble des vecteurs invariants et de direction son noyau.

\textbf{e.} \underline{Base :} $f(\vec{u}) = \vec{u} \Leftrightarrow 3x - 2y = x$ et $3x - 2y = y$, soit $x - y = 0$ : droite engendrée par $(1, 1)$. \underline{Direction :} $f(\vec{u}) = \vec{0} \Leftrightarrow 3x - 2y = 0$ : droite engendrée par $(2, 3)$.

\textbf{4. a.} $\dfrac{\frac{1}{x}}{\ln x} = \dfrac{1}{x}\times\dfrac{1}{\ln x} = \dfrac{1}{x\ln x}$.

\textbf{b.} $\dfrac{1/x}{\ln x}$ est de la forme $\dfrac{u'(x)}{u(x)}$ avec $u(x) = \ln x$, donc
\[ I = \int_e^3 \frac{1}{x\ln x}\,dx = \big[\ln(\ln x)\big]_e^3 = \ln(\ln 3) - \ln(\ln e) = \ln\ln 3. \]

\exo{4}{}
On note $(x_i, n_{i\bullet})$ et $(y_j, n_{\bullet j})$ les distributions marginales, $N = \sum_i n_{i\bullet} = \sum_j n_{\bullet j}$.

\textbf{1.} Lois marginales :
\begin{center}\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|c|c|}\hline $X$ & $-1$ & $0$ & $2$ \\\hline $n_{i\bullet}$ & 3 & 5 & 2 \\\hline\end{tabular}\qquad
\begin{tabular}{|c|c|c|}\hline $Y$ & $-1$ & $1$ \\\hline $n_{\bullet j}$ & 6 & 4 \\\hline\end{tabular}
\end{center}

\textbf{2.} $\overline{X} = \dfrac{3(-1) + 5(0) + 2(2)}{10} = 0{,}1$ ; \quad $\overline{Y} = \dfrac{6(-1) + 4(1)}{10} = -0{,}2$.

\textbf{3.} Inertie du nuage :
\[ I_G = \sum_i n_{i\bullet}(x_i - \overline{X})^2 + \sum_j n_{\bullet j}(y_j - \overline{Y})^2 = 3(1{,}1)^2 + 5(0{,}1)^2 + 2(1{,}9)^2 + 6(0{,}8)^2 + 4(1{,}2)^2 = 20{,}5. \]

\textbf{4.} $\rho_{X,Y} = \dfrac{\mathrm{Cov}(X,Y)}{\sqrt{V(X)}\sqrt{V(Y)}}$ avec
\[ V(X) = \frac{3(-1)^2 + 5(0)^2 + 2(2)^2}{10} - 0{,}1^2 = 1{,}09 \;;\quad V(Y) = \frac{6(-1)^2 + 4(1)^2}{10} - (-0{,}2)^2 = 0{,}96, \]
\[ \mathrm{Cov}(X,Y) = \frac{2 - 1 + 0 + 0 - 2 + 2}{10} - 0{,}1\times(-0{,}2) = 0{,}12. \]
D'où $\rho_{X,Y} = \dfrac{0{,}12}{\sqrt{1{,}09}\,\sqrt{0{,}96}} = 0{,}117$.
