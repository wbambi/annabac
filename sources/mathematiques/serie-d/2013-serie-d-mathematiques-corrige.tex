\exo{1}{}
On note $(x_i, n_{i\bullet})$ les couples de la distribution marginale de $X$, et $(y_j, n_{\bullet j})$ ceux de $Y$. On a $\sum_i n_{i\bullet} = \sum_j n_{\bullet j} = N = 19$.

\textbf{1.}
\begin{center}\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|c|c|}\hline $X$ & $-2$ & $-1$ & $0$ \\\hline $n_{i\bullet}$ & 6 & 8 & 5 \\\hline\end{tabular}\qquad
\begin{tabular}{|c|c|c|c|}\hline $Y$ & $-1$ & $0$ & $2$ \\\hline $n_{\bullet j}$ & 9 & 6 & 4 \\\hline\end{tabular}
\end{center}

\textbf{2.} $\overline{X} = \dfrac{1}{N}\sum_{i=1}^3 n_{i\bullet}x_i = \dfrac{6(-2) + 8(-1) + 5(0)}{19} = -\dfrac{20}{19}$ ; \quad $\overline{Y} = \dfrac{9(-1) + 6(0) + 4(2)}{19} = -\dfrac{1}{19}$. Le point moyen est $G\left(-\dfrac{20}{19}, -\dfrac{1}{19}\right)$.

\textbf{3.} L'équation de régression de $Y$ en $X$ est $Y = aX + b$ où $a = \dfrac{\mathrm{Cov}(X,Y)}{V(X)}$ et $b = \overline{Y} - a\overline{X}$.
\[ \mathrm{Cov}(X,Y) = \frac{1}{N}\sum_{i}\sum_{j} n_{ij}x_i y_j - \overline{X}\,\overline{Y} = \frac{8 + 0 - 8 + 3}{19} - \frac{20}{19^2} = \frac{37}{361} \]
\[ V(X) = \frac{1}{N}\sum_i n_{i\bullet}x_i^2 - \overline{X}^2 = \frac{6(-2)^2 + 8(-1)^2 + 5(0)^2}{19} - \left(\frac{20}{19}\right)^2 = \frac{208}{361} \]
Ainsi $a = \dfrac{37/361}{208/361} = \dfrac{37}{208}$ et $b = -\dfrac{1}{19} - \dfrac{37}{208}\cdot\dfrac{-20}{19} = \dfrac{7}{52}$. D'où $Y = \dfrac{37}{208}X + \dfrac{7}{52}$.

\textbf{4.} $\rho_{X,Y} = \dfrac{\mathrm{Cov}(X,Y)}{\sqrt{V(X)}\sqrt{V(Y)}}$ avec $V(Y) = \dfrac{9(-1)^2 + 6(0)^2 + 4(2)^2}{19} - \left(\dfrac{1}{19}\right)^2 = \dfrac{474}{361}$. D'où $\rho_{X,Y} = \dfrac{37/361}{\sqrt{208/361}\,\sqrt{474/361}} = 0{,}117$.

\exo{2}{}
\textbf{1.} Posons $Z = r\,e^{i\theta}$.
\[ Z^3 = -8 \Longleftrightarrow r^3 e^{i3\theta} = 8\,e^{i\pi} \Longleftrightarrow \begin{cases} r = 2 \\ \theta_k = \dfrac{\pi}{3} + \dfrac{2k\pi}{3},\ k \in \Z \end{cases} \]
D'où $z_0 = 2e^{i\frac{\pi}{3}} = 1 + i\sqrt{3}$ ; \quad $z_1 = 2e^{i\pi} = -2$ ; \quad $z_2 = 2e^{i\frac{5\pi}{3}} = 1 - i\sqrt{3}$.

\textbf{2. a.} $U = \dfrac{Z_C - Z_A}{Z_B - Z_A} = \dfrac{-2i\sqrt{3}}{-3 - i\sqrt{3}} = \dfrac{1}{2} + i\dfrac{\sqrt{3}}{2}$. D'où $|U| = 1$ et $\arg(U) = \dfrac{\pi}{3}\ [2\pi]$.

\textbf{b.} Comme $|U| = 1$, alors $AC = AB$. De plus $(\vv{AB}, \vv{AC}) \equiv \arg(U) \equiv \dfrac{\pi}{3}\ [2\pi]$. Le triangle $ABC$ est isocèle en $A$ avec un angle de $\dfrac{\pi}{3}$ : c'est un triangle équilatéral.

\textbf{3. a.} Soit $Z' = aZ + b$ l'expression de la rotation $S$.
\[ \begin{cases} S(A) = C \\ S(C) = B \end{cases} \Longleftrightarrow \begin{cases} Z_C = aZ_A + b \\ Z_B = aZ_C + b \end{cases} \Rightarrow a = \frac{Z_C - Z_B}{Z_A - Z_C} = \frac{3 - i\sqrt{3}}{2i\sqrt{3}} = -\frac{1}{2} - i\frac{\sqrt{3}}{2} \]
et $b = Z_C - aZ_A = 1 - i\sqrt{3} - \left(-\tfrac{1}{2} - i\tfrac{\sqrt{3}}{2}\right)(1 + i\sqrt{3}) = 0$. D'où $Z' = \left(-\dfrac{1}{2} - i\dfrac{\sqrt{3}}{2}\right)Z$.

\textbf{b.} $Z' = e^{i\frac{4\pi}{3}}Z$ : $S$ est la rotation de centre $O$ (affixe 0) et d'angle $\dfrac{4\pi}{3}$.

\partieT{A}
\textbf{1.} $g$ est dérivable sur $]0, +\infty[$ et $g'(x) = -\dfrac{2x^2 + 1}{x} < 0$ pour tout $x > 0$. Avec $\lim\limits_{x\to 0^+} g(x) = +\infty$ et $\lim\limits_{x\to+\infty} g(x) = -\infty$ :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccc|}
\hline $x$ & $0$ & & $+\infty$ \\\hline $g'(x)$ & $\|$ & $-$ & \\\hline $g$ & $+\infty$ & $\searrow$ & $-\infty$ \\\hline
\end{tabular}
\end{center}

\textbf{2.} $g$ est strictement décroissante sur $]0;+\infty[$ et $g(1) = 0$. Donc $g(x) > 0$ sur $]0;1[$ et $g(x) < 0$ sur $]1;+\infty[$.

\partieT{B}
\textbf{1.} $x \mapsto 1 + x - e^{1-x}$ existe sur $]-\infty;1]$ ; $x \mapsto \dfrac{2x - x^2 + \ln x}{x}$ existe sur $]1;+\infty[$. Donc $E_f = \R$.

\textbf{2. a.} \underline{Continuité en 1 :} $\lim\limits_{x\to 1^-} f(x) = f(1) = 1$ et $\lim\limits_{x\to 1^+} f(x) = 1$, donc $f$ est continue en 1.

\underline{Dérivabilité en 1 :} $\lim\limits_{x\to 1^-}\dfrac{f(x) - f(1)}{x - 1} = 1 + \lim\limits_{u\to 0^+}\dfrac{e^u - 1}{u} = 2$ (avec $u = 1 - x$) ; $\lim\limits_{x\to 1^+}\dfrac{f(x) - f(1)}{x - 1} = -1 + \lim\limits_{x\to 1^+}\dfrac{\ln x}{x(x-1)} = 0$. Les deux limites diffèrent : $f$ n'est pas dérivable en 1.

\textbf{b.} Pour tout $x \in ]1;+\infty[$, $f'(x) = \dfrac{g(x)}{x^2}$.

\textbf{3.} $f$ est dérivable sur $]-\infty;1[\,\cup\,]1;+\infty[$ : $f'(x) = 1 + e^{1-x}$ sur $]-\infty;1[$ ; $f'(x) = \dfrac{g(x)}{x^2}$ sur $]1;+\infty[$.

\underline{Signe :} $f'(x) > 0$ sur $]-\infty;1[$ ; $f'(x)$ est du signe de $g(x)$, donc $f'(x) < 0$ sur $]1;+\infty[$. Avec $\lim\limits_{x\to-\infty} f = -\infty$ et $\lim\limits_{x\to+\infty} f = -\infty$ :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline $x$ & $-\infty$ & & $1$ & & $+\infty$ \\\hline $f'(x)$ & & $+$ & $\|$ & $-$ & \\\hline $f$ & $-\infty$ & $\nearrow$ & $1$ & $\searrow$ & $-\infty$ \\\hline
\end{tabular}
\end{center}

\textbf{4.} \underline{$(\Delta)$ asymptote :} $\lim\limits_{x\to+\infty}\big(f(x) - (2 - x)\big) = \lim\limits_{x\to+\infty}\dfrac{\ln x}{x} = 0$, donc $(\Delta) : y = -x + 2$ est asymptote en $+\infty$. Comme $f(x) - (2-x) = \dfrac{\ln x}{x} > 0$ pour $x > 1$, $(\mathcal{C})$ est au-dessus de $(\Delta)$.

\textbf{5.} Tangente en $x = 0$ : $y = f'(0)x + f(0)$, soit $(T) : y = (1 + e)x + 1 - e$.

\textbf{6.} $\lim\limits_{x\to-\infty}\dfrac{f(x)}{x} = +\infty$ : branche parabolique de direction $(Oy)$ en $-\infty$. En $+\infty$, $(\Delta) : y = 2 - x$ est asymptote oblique.
