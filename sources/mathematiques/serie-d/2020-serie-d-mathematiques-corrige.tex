\exo{1}{}
\textbf{1.} $i\overline{z_2} = 1 - 2i \Leftrightarrow -i z_2 = 1 + 2i \Leftrightarrow z_2 = -2 + i$. Puis $z_1 = -3 + i - z_2 = -1$ et $z_3 = \dfrac{-1 - 2i}{-2 + i} = i$. D'où $z_1 = -1$, $z_2 = -2 + i$, $z_3 = i$.

\textbf{2. a.} $P(i) = i^3 + (3 - 2i)i^2 + (1 - 4i)i - 1 - 2i = 0$.

\textbf{b.} Dans $(z - i)(z - z_0)(z + 2 - i)$, le terme constant est $i z_0(2 - i)$ ; par identification $z_0(1 + 2i) = -1 - 2i$, d'où $z_0 = \dfrac{-1 - 2i}{1 + 2i} = -1$.

\textbf{c.} $P(z) = 0 \Leftrightarrow (z - i)(z + 1)(z + 2 - i) = 0$, soit $z \in \{i,\ -1,\ -2 + i\}$.

\textbf{3. a.} $R$ : $z' - z_A = a(z - z_A)$ avec $|a| = 1$. Comme $R(B) = C$, $a = \dfrac{z_C - z_A}{z_B - z_A} = \dfrac{1 + i}{-1 + i} = -i$. D'où $z' = -i(z + 1) - 1 = -iz - i - 1$.

\textbf{b.} $a = -i = e^{-i\frac{\pi}{2}}$ : l'angle de $R$ est $-\dfrac{\pi}{2}$.

\exo{2}{}
\textbf{1. a.} $\det_{(\vi,\vj)}(\vec{u}, \vec{v}) = \begin{vmatrix} 1 & -1 \\ 1 & 2 \end{vmatrix} = 3 \ne 0$, donc $\{\vec{u}, \vec{v}\}$ est une base de $\R^2$.

\textbf{b.} De $\begin{cases} \vi + \vj = \vec{u} \\ -\vi + 2\vj = \vec{v} \end{cases}$ on tire $\vi = \dfrac{2}{3}\vec{u} - \dfrac{1}{3}\vec{v}$ et $\vj = \dfrac{1}{3}\vec{u} + \dfrac{1}{3}\vec{v}$.

\textbf{2. a.} $f(\vi) = \dfrac{2}{3}f(\vec{u}) - \dfrac{1}{3}f(\vec{v}) = \dfrac{2}{3}\vec{u} + \dfrac{1}{3}\vec{v} = \dfrac{1}{3}\vi + \dfrac{4}{3}\vj$ ; $f(\vj) = \dfrac{1}{3}\vec{u} - \dfrac{1}{3}\vec{v} = \dfrac{2}{3}\vi - \dfrac{1}{3}\vj$.

\textbf{b.} $f\circ f(\vi) = \vi$ et $f\circ f(\vj) = \vj$.

\textbf{c.} $f\circ f = \mathrm{id}$ sur la base $(\vi,\vj)$ : $f$ est une symétrie vectorielle.

\textbf{d.} Comme $f(\vec{u}) = \vec{u}$ et $f(\vec{v}) = -\vec{v}$ : la base $\mathscr{E}$ est engendrée par $\vec{u}$, la direction $\mathscr{D}$ par $\vec{v}$.

\exo{3}{}
\partieT{A}
\textbf{1. a.} $\forall x \in \R$, $h'(x) = (x - 2)e^{2-x}$. \textbf{b.} $h'$ s'annule en $x = 2$, du signe de $x - 2$.

\textbf{c.}
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline $x$ & $-\infty$ & & $2$ & & $+\infty$ \\\hline $h'(x)$ & & $-$ & $0$ & $+$ & \\\hline $h$ & $+\infty$ & $\searrow$ & $0$ & $\nearrow$ & $1$ \\\hline
\end{tabular}
\end{center}

\textbf{2.} $h$ décroît de $]-\infty;2]$ sur $[0;+\infty[$ et croît de $[2;+\infty[$ sur $[0;1[$. Donc $h(x) \ge 0$ pour tout $x \in \R$.

\partieT{B}
\textbf{1.} $\lim\limits_{x\to+\infty} f(x) = \lim\limits_{x\to+\infty} x\cdot\underbrace{(1 + e^{2-x})}_{\to 1} = +\infty$.

\textbf{2. a.} $\forall x \in \R$, $f'(x) = (1 + e^{2-x}) + x(-e^{2-x}) = (1 - x)e^{2-x} + 1 = h(x) \ge 0$.

\textbf{b.} $f$ est strictement croissante ($f'(2) = h(2) = 0$, $f(2) = 4$), $\lim\limits_{-\infty} f = -\infty$, $\lim\limits_{+\infty} f = +\infty$ :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline $x$ & $-\infty$ & & $2$ & & $+\infty$ \\\hline $f'(x)$ & & $+$ & $0$ & $+$ & \\\hline $f$ & $-\infty$ & $\nearrow$ & $4$ & $\nearrow$ & $+\infty$ \\\hline
\end{tabular}
\end{center}

\textbf{3. a.} $f$ est continue strictement croissante sur $\R$ : elle admet une bijection réciproque $f^{-1}$ définie sur $\R$.

\textbf{b.} $f$ et $f^{-1}$ ont même sens de variation. $f$ a une tangente horizontale en $x = 2$, donc $f^{-1}$ a une tangente verticale en $f(2) = 4$ : $f^{-1}$ n'est pas dérivable en 4.
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccc|}
\hline $x$ & $-\infty$ & $4$ & $+\infty$ \\\hline $(f^{-1})'(x)$ & $+$ & $\|$ & $+$ \\\hline $f^{-1}$ & $-\infty$ & $2$ & $+\infty$ \\\hline
\end{tabular}
\end{center}

\textbf{4. a.} $\lim\limits_{x\to-\infty}\dfrac{f(x)}{x} = \lim\limits_{x\to-\infty}(1 + e^{2-x}) = +\infty$ : branche parabolique de direction $(Oy)$ en $-\infty$.

\textbf{b.} $\lim\limits_{x\to+\infty}(f(x) - x) = \lim\limits_{x\to+\infty} x e^{2-x} = 0$ : la droite $(\Delta) : y = x$ est asymptote en $+\infty$.

\textbf{c.} $f(x) - x = x e^{2-x}$ : $(\mathcal{C})$ est au-dessus de $(\Delta)$ sur $\R_+$ et en dessous sur $\R_-$.
