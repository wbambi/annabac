\textbf{1.} L'ﺣ۸quation $(E) : Z^2 - (2ie^{i\theta}\cos\theta)Z - e^{i2\theta} = 0$ admet pour discriminant rﺣ۸duit :
\[ \Delta' = (ie^{i\theta}\cos\theta)^2 - 1 \times (-e^{2i\theta}) = e^{2i\theta}(-\cos^2\theta + 1) = (e^{i\theta}\sin\theta)^2. \]
On en dﺣ۸duit que les racines de $(E)$ sont :
\begin{align*}
Z_1 &= ie^{i\theta}\cos\theta - e^{i\theta}\sin\theta = e^{i\theta}\,i(\cos\theta + i\sin\theta) = e^{i\theta}e^{i\frac{\pi}{2}}e^{i\theta} = e^{i\left(\frac{\pi}{2} + 2\theta\right)} \\
Z_2 &= ie^{i\theta}\cos\theta + e^{i\theta}\sin\theta = e^{i\theta}\,i(\cos\theta - i\sin\theta) = e^{i\theta}e^{i\frac{\pi}{2}}e^{-i\theta} = e^{i\frac{\pi}{2}}
\end{align*}

\textbf{2. a.} $\arg\left(\dfrac{Z_B - Z_0}{Z_A - Z_0}\right) = \arg\left(\dfrac{e^{i\left(\frac{\pi}{2} + 2\theta\right)}}{e^{i\frac{\pi}{2}}}\right) = \arg(e^{i2\theta}) \equiv 2\theta\ [2\pi]$.

\textbf{b.} Comme $(\vv{OA}, \vv{OB}) \equiv \arg\left(\dfrac{Z_B - Z_0}{Z_A - Z_0}\right)\ [2\pi]$, alors $2\theta \equiv \dfrac{\pi}{2}\ [2\pi]$. D'oﺣﺗ $\theta \equiv \dfrac{\pi}{4}\ [\pi]$. L'ensemble des valeurs de $\theta$ est $\left\{\dfrac{\pi}{4} + k\pi\,;k \in \Z\right\}$.

\textbf{c.}
\begin{align*}
Z_A + Z_B &= e^{i\frac{\pi}{2}} + e^{i\left(\frac{\pi}{2} + 2\theta\right)} = e^{i\frac{\pi}{2}}(1 + e^{i2\theta}) = e^{i\frac{\pi}{2}}(1 + \cos 2\theta + i\sin 2\theta) \\
&= e^{i\frac{\pi}{2}}(2\cos^2\theta + i\,2\sin\theta\cos\theta) = 2e^{i\frac{\pi}{2}}\cos\theta(\cos\theta + i\sin\theta) = 2\cos\theta\,e^{i\left(\frac{\pi}{2} + \theta\right)}
\end{align*}
On a $|Z_A + Z_B| = |2\cos\theta| = 2\cos\theta$ car $0 \le \theta < \dfrac{\pi}{2}$. On en dﺣ۸duit que $2\cos\theta\,e^{i\left(\frac{\pi}{2} + \theta\right)}$ est la forme exponentielle de $Z_A + Z_B$. Donc la forme exponentielle de $\overline{Z_A + Z_B}$ est $2\cos\theta\,e^{-i\left(\frac{\pi}{2} + \theta\right)}$.

\textbf{1.} La fonction $x \mapsto e^{-x}$ est dﺣ۸finie sur $\R$ et $x \mapsto x^{n+1}$ est dﺣ۸finie sur $\R$, donc $x \mapsto e^{-x}x^{n+1}$ est dﺣ۸finie sur $\R$. D'oﺣﺗ $E_{f_n} = \R$.

\textbf{2. a.} $f_n$ est dﺣ۸rivable sur $\R$. $\forall x \in \R$, $f_n'(x) = (-e^{-x})x^{n+1} + e^{-x}(n+1)x^n = (-x + n + 1)e^{-x}x^n$. L'entier $n$ ﺣ۸tant impair, $x^n$ est de mﺣ۹me signe que $x$. D'oﺣﺗ le tableau de signes :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $-\infty$ & & $0$ & $n+1$ & $+\infty$ \\
\hline
$-x + n + 1$ & & $+$ & $+$ \quad $0$ & $-$ & \\
\hline
$e^{-x}x^n$ & & $-$ & $0$ \quad $+$ & $+$ & \\
\hline
$f_n'(x)$ & & $-$ & $0$ \quad $+$ \quad $0$ & $-$ & \\
\hline
\end{tabular}
\end{center}

\textbf{b.} $n + 1$ ﺣ۸tant pair, $\lim\limits_{x\to-\infty} x^{n+1} = +\infty$. D'oﺣﺗ $\lim\limits_{x\to-\infty} e^{-x}x^{n+1} = +\infty$ et $\lim\limits_{x\to+\infty} e^{-x}x^{n+1} = 0$.

\textbf{3. a.} En posant $u'(x) = e^{-x}$, $v(x) = x^{n+2}$, donc $u(x) = -e^{-x}$, $v'(x) = (n+2)x^{n+1}$, on a en intﺣ۸grant par parties :
\[ I_{n+1,p} = \int_0^p e^{-x}x^{n+2}\,dx = \left[-e^{-x}x^{n+2}\right]_0^p + (n+2)\int_0^p e^{-x}x^{n+1}\,dx = -e^{-p}p^{n+2} + (n+2)I_{n,p}. \]
Par passage ﺣ  la limite : $\lim\limits_{p\to+\infty} I_{n+1,p} = \underbrace{\lim\limits_{p\to+\infty}(-e^{-p}p^{n+2})}_{=0} + (n+2)\lim\limits_{p\to+\infty} I_{n,p}$. D'oﺣﺗ $J_{n+1} = (n+2)J_n$.

\textbf{b.} $J_n = (n+1)J_{n-1} = (n+1) \times n J_{n-2} = \cdots = (n+1) \times n \times \cdots \times 2 \times J_0 = (n+1)!\,J_0$.
