\textbf{1.}
\begin{align*}
2x - 7y = 3 &\Longleftrightarrow 2x - 3 = 7y,\ y \in \Z \\
&\Longleftrightarrow 2x - 3 \text{ est un multiple de } 7 \Longleftrightarrow 2x - 3 \equiv 0\ [7] \Longleftrightarrow 2x \equiv 3\ [7]
\end{align*}
D'où les équations $(E_0)$ et $(E_1)$ sont équivalentes.

\textbf{2.} \underline{Solution particulière de $2x \equiv 3\ [7]$ :} 2 et $-7$ étant premiers entre eux, de l'égalité $7 = 2 \times 3 + 1$ on déduit $2 \times (-3) - 7 \times (-1) = 1$. En multipliant par 3 : $2 \times (-9) - 7 \times (-3) = 3$. On en déduit que $(-9, -3)$ est une solution particulière de $(E_1)$ et $2 \times (-9) \equiv 3\ [7]$.

\underline{Ensemble des solutions de $(E_1)$ :}
\[ \begin{cases} 2x \equiv 3\ [7] \\ 2 \times (-9) \equiv 3\ [7] \end{cases} \Longleftrightarrow 2x \equiv 2 \times (-9)\ [7] \Longleftrightarrow 2(x + 9) \equiv 0\ [7] \Longleftrightarrow 7 \mid 2(x + 9). \]
Comme 7 est premier avec 2, d'après Gauss, 7 divise $x + 9$. Il existe donc $k \in \Z$ tel que $x + 9 = 7k$. D'où $x = -9 + 7k$. Les solutions de $(E_1)$ sont $\{-9 + 7k\,;k \in \Z\}$.

\begin{rappel}
Théorème de Gauss : si un entier $a$ divise un produit $bc$ et si $a$ est
premier avec $b$, alors $a$ divise $c$. C'est lui qui permet de passer de
$7 \mid 2(x+9)$ à $7 \mid x+9$, car 7 et 2 sont premiers entre eux.
\end{rappel}

\begin{methode}
Pour résoudre une congruence $ax \equiv b\ [n]$ : \textbf{1)} chercher une
solution particulière $x_0$ (essais successifs, ou identité de Bézout comme
ici) ; \textbf{2)} soustraire les deux congruences pour obtenir
$a(x - x_0) \equiv 0\ [n]$ ; \textbf{3)} conclure avec le théorème de Gauss
lorsque $a$ et $n$ sont premiers entre eux : $x \equiv x_0\ [n]$.
\end{methode}

\textbf{3.} En remplaçant $x$ par $-9 + 7k$ dans $(E_0) : 2x - 7y = 3$, on obtient $y = -3 + 2k$. L'ensemble des solutions de $(E_0)$ est $\{(-9 + 7k\,,\,-3 + 2k)\,;k \in \Z\}$.

\textbf{4. a.} $A = \{1\,;5\,;19\,;95\}$.

\textbf{b.} Soit $(x, y) \in \Z \times \Z$ vérifiant le système $\begin{cases} 2x - 7y = 3 \\ xy = 95 \end{cases}$. Comme $2x - 7y = 3 > 0$, alors $x > y$. En utilisant 4.a., les solutions $(x, y)$ de $xy = 95$ vérifiant $x > y$ appartiennent à $I = \{(95, 1), (-1, -95), (-5, -19), (19, 5)\}$.

\underline{Couple $(x, y) \in I$ solution de $2x - 7y = 3$ :}
\begin{itemize}
  \item $2 \times 95 - 7 \times 1 \ne 3$ : $(95, 1)$ n'est pas solution.
  \item $2(-1) - 7(-95) \ne 3$ : $(-1, -95)$ n'est pas solution.
  \item $2(-5) - 7(-19) \ne 3$ : $(-5, -19)$ n'est pas solution.
  \item $2 \times 19 - 7 \times 5 = 3$ : $(19, 5)$ est solution.
\end{itemize}
D'où $(19, 5)$ est le seul couple solution du système.

\begin{piege}
Un couple qui vérifie $xy = 95$ n'est pas forcément solution du système :
chaque candidat doit aussi être testé dans $2x - 7y = 3$, sans en oublier
(y compris les couples négatifs), avant de conclure.
\end{piege}

\underline{Autre méthode :} $(x, y)$ vérifie $2x - 7y = 3$, donc d'après 3., $(x, y) = (-9 + 7k\,,\,-3 + 2k)$, $k \in \Z$. Ce couple vérifie $xy = 95$ si $(-9 + 7k)(-3 + 2k) = 95$, soit $14k^2 - 39k - 68 = 0$, qui admet dans $\Z$ une seule solution $k = 4$. En remplaçant : $(-9 + 28\,,\,-3 + 8) = (19, 5)$. Donc $(19, 5)$ est le seul couple solution.

\textbf{1. a.} $f$ est dérivable sur $]0\,;+\infty[$. $\forall x \in ]0\,;+\infty[$, $f'(x) = \dfrac{x^2 - 4}{2x} = \dfrac{(x - 2)(x + 2)}{2x}$.

\textbf{b.} $f'$ est du signe de $x \mapsto x - 2$ sur $]0\,;+\infty[$.

\underline{Limites :} $\lim\limits_{x\to 0^+} f(x) = -\dfrac{1}{4} - 2\lim\limits_{x\to 0^+}\ln x = +\infty$. $\lim\limits_{x\to+\infty} f(x) = \lim\limits_{x\to+\infty} x^2\left(\dfrac{1}{4} - \dfrac{1}{4x^2} - \dfrac{2\ln x}{x^2}\right) = +\infty$.

\underline{Tableau de variation :}

\textbf{c.} $f(3) \approx -0{,}197$ et $f(4) \approx 0{,}977$. La fonction $f$ est continue et strictement croissante sur $]3\,;4[$, et $f(3) \cdot f(4) < 0$. D'après le théorème des valeurs intermédiaires, il existe un unique réel $\alpha \in [3\,;4]$ tel que $f(\alpha) = 0$.

\textbf{2. a.} Remarquons d'abord que si $x \ge 3$, $8\ln x + 1 > 0$.
\begin{align*}
f(x) = 0 &\Longleftrightarrow \frac{x^2 - 1}{4} = 2\ln x \Longleftrightarrow x^2 = 8\ln x + 1 \\
&\Longleftrightarrow x = \sqrt{8\ln x + 1} \quad (\text{le cas } x = -\sqrt{8\ln x + 1} \text{ est impossible car } x \ge 3)
\end{align*}
D'où les équations $f(x) = 0$ et $g(x) = x$ sont équivalentes sur $[3\,;+\infty[$.

\textbf{b.} $g$ est continue et dérivable sur $[3\,;4]$, et $|g'(x)| \le \dfrac{4}{9}$ pour tout $x \in [3\,;4]$. D'après le théorème de l'inégalité des accroissements finis, $\forall x, y \in [3\,;4]$, $|g(x) - g(y)| \le \dfrac{4}{9}|x - y|$. En prenant $y = \alpha$ et puisque $g(\alpha) = \alpha$ : $\forall x \in [3\,;4]$, $|g(x) - \alpha| \le \dfrac{4}{9}|x - \alpha|$.

\textbf{3. a.} Soit $\mathcal{P}_n$ : $u_n \in [3\,;4]$. \underline{Initialisation :} $u_0 = 3 \in [3\,;4]$, et d'après 2.b. $u_1 = g(u_0) \in [3\,;4]$. \underline{Hérédité :} si $u_n \in [3\,;4]$, alors d'après 2.b. $g(u_n) = u_{n+1} \in [3\,;4]$. Donc $\mathcal{P}_n$ est vraie pour tout $n \in \N$.

\textbf{b.} Soit $n \in \N$. En appliquant l'inégalité de 2.b. en $x = u_n$ : $|g(u_n) - \alpha| \le \dfrac{4}{9}|u_n - \alpha|$. Or $g(u_n) = u_{n+1}$, d'où $|u_{n+1} - \alpha| \le \dfrac{4}{9}|u_n - \alpha|$ pour tout $n \in \N$.

\textbf{c.} Soit $\mathcal{P}_n$ : $|u_n - \alpha| \le \left(\dfrac{4}{9}\right)^n$. \underline{Initialisation :} comme $3 \le \alpha \le 4$, $-1 \le u_0 - \alpha \le 0$ donc $|u_0 - \alpha| \le 1 = \left(\dfrac{4}{9}\right)^0$. \underline{Hérédité :} d'après 3.b. et l'hypothèse, $|u_{n+1} - \alpha| \le \dfrac{4}{9}|u_n - \alpha| \le \dfrac{4}{9}\left(\dfrac{4}{9}\right)^n = \left(\dfrac{4}{9}\right)^{n+1}$. Donc $\mathcal{P}_n$ est vraie pour tout $n \in \N$.

\textbf{d.} $\forall n \in \N$, $0 \le |u_n - \alpha| \le \left(\dfrac{4}{9}\right)^n$. Par passage à la limite, $\lim\limits_{n\to+\infty}|u_n - \alpha| = 0$, donc $\lim\limits_{n\to+\infty} u_n = \alpha$.

\textbf{e.} Soit $n \in \N$ tel que $\left(\dfrac{4}{9}\right)^n \le 10^{-2}$. Par croissance du logarithme, $n\ln\dfrac{4}{9} \le -2\ln 10$, d'où $n \ge \dfrac{2\ln 10}{\ln 9 - \ln 4} \approx 5{,}67$. Donc $n_0 = 6$.
