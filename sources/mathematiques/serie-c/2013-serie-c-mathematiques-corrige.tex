\exo{1}{}

\textbf{1.} En remplaçant $z$ par $-1$ dans l'équation $(E)$, on vérifie que :
\[ (-1)^3 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)(-1)^2 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)(-1) + 1 = 0. \]

\textbf{2.} Soit $z_0$ une solution de $(E)$. Le nombre $z_0$ est nécessairement non nul puisque 0 n'est pas solution de $(E)$.
\[ \left(\tfrac{1}{z_0}\right)^3 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)\left(\tfrac{1}{z_0}\right)^2 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)\left(\tfrac{1}{z_0}\right) + 1 = \frac{1 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)z_0 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)z_0^2 + z_0^3}{z_0^3} = \frac{0}{z_0^3} = 0. \]

\textbf{3.} L'équation $(E') : z^2 - \left(\tfrac{3}{2} + \tfrac{1}{2}i\right)z + 1$ admet pour discriminant $\Delta = -2 + \dfrac{3}{2}i$.

Cherchons $u = x + iy$ tel que $u^2 = -2 + \dfrac{3}{2}i$, soit $x^2 - y^2 + 2ixy = -2 + \dfrac{3}{2}i$. Par identification : $x^2 - y^2 = -2$ et $xy = \dfrac{3}{4}$. D'autre part, $|u|^2 = \left|-2 + \tfrac{3}{2}i\right|$ donne $x^2 + y^2 = \dfrac{5}{2}$. On obtient le système :
\[ \begin{cases} x^2 - y^2 = -2 & (1) \\ x^2 + y^2 = \dfrac{5}{2} & (2) \\ xy = \dfrac{3}{4} & (3) \end{cases} \]
En additionnant (1) et (2), $x = -\dfrac{1}{2}$ ou $x = \dfrac{1}{2}$ ; de même $y = -\dfrac{3}{2}$ ou $y = \dfrac{3}{2}$. L'équation (3) indique que $x$ et $y$ sont de même signe. D'où $\Delta = \left(\dfrac{1}{2} + \dfrac{3}{2}i\right)^2$. On en déduit que les racines de $(E')$ sont : $z'_0 = 1 + i$ et $z''_0 = \dfrac{1}{2} - \dfrac{1}{2}i$.

\textbf{4.} En remarquant que $(z + 1)\left(z^2 - \left(\tfrac{3}{2} + \tfrac{1}{2}i\right)z + 1\right) = z^3 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)z^2 - \left(\tfrac{1}{2} + \tfrac{1}{2}i\right)z + 1$, on en déduit que les solutions de $(E)$ sont : $\left\{-1,\ 1 + i,\ \dfrac{1}{2} - \dfrac{1}{2}i\right\}$.

\[ \overline{X} = \frac{1}{N}\sum_{i=1}^3 n_{i\bullet}x_i = \frac{5(-2) + 7(-1) + 2(0)}{14} = -\frac{17}{14}, \quad \overline{Y} = \frac{6(-1) + 3(0) + 5(2)}{14} = \frac{2}{7}. \]

\textbf{3.}
\[ V(X) = \frac{1}{N}\sum_{i=1}^3 n_{i\bullet}x_i^2 - \overline{X}^2 = \frac{1}{14}\left[5(-2)^2 + 7(-1)^2 + 2(0)^2\right] - \left(-\frac{17}{14}\right)^2 = \frac{89}{196}. \]
\[ V(Y) = \frac{1}{N}\sum_{j=1}^3 n_{\bullet j}y_j^2 - \overline{Y}^2 = \frac{1}{14}\left[6(-1)^2 + 3(0)^2 + 5(2)^2\right] - \left(\frac{2}{7}\right)^2 = \frac{87}{49}. \]

\textbf{4.}
\[ \mathrm{Cov}(X, Y) = \frac{1}{N}\sum_{i=1}^3\sum_{j=1}^3 n_{ij}x_i y_j - \overline{X}\cdot\overline{Y} = \frac{1}{14}\big(3 \times 2 + 2 \times 1 + 2 \times (-4) + 2 \times (-2)\big) - \left(-\frac{17}{14}\right) \times \frac{2}{7} = \frac{3}{49}. \]

\textbf{5.} Le coefficient de corrélation linéaire entre $X$ et $Y$ est :
\[ \rho_{X,Y} = \frac{\mathrm{Cov}(X, Y)}{\sqrt{V(X)}\sqrt{V(Y)}} = 0{,}068. \]
