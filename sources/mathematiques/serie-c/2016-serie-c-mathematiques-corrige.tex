\textbf{1.} L'algorithme d'Euclide donne : $3024 = 2688 \times 1 + 336$, $2688 = 336 \times 8 + 0$. Le PGCD, dernier reste non nul, est donc égal à 336.

Comme $\mathrm{PGCD}(2688;3024) = 336$, il existe des couples $(u, v)$ dans $\Z \times \Z$ tels que $2688u + 3024v = 336$. En multipliant par $-10$ : $2688(-10u) + 3024(-10v) = -3360$. On en déduit que les couples $(-10u, -10v)$ sont solutions de $(E)$. Donc $(E)$ admet des solutions dans $\Z \times \Z$.

\textbf{2.} $2688x + 3024y = -3360 \Longleftrightarrow \dfrac{2688}{336}x + \dfrac{3024}{336}y = -\dfrac{3360}{336} \Longleftrightarrow 8x + 9y = -10$. On en déduit que $(E)$ et $(E_1)$ sont équivalentes.

\textbf{3. a.}
\begin{align*}
8x + 9y = -10 &\Longleftrightarrow 8x + 10 = 9(-y),\ y \in \Z \\
&\Longleftrightarrow 8x + 10 \text{ est un multiple de } 9 \Longleftrightarrow 8x + 10 \equiv 0\ [9] \Longleftrightarrow 8x \equiv -10\ [9]
\end{align*}
D'où $(E_1)$ et $(E_2)$ sont équivalentes.

\textbf{b.} 1 est une solution particulière de $(E_2) : 8x \equiv -10\ [9]$, car $8 \times 1 = -10 + 9 \times 2 \equiv -10\ [9]$. Soit $x$ une solution de $(E_2)$ :
\[ \begin{cases} 8x \equiv -10\ [9] \\ 8 \times 1 \equiv -10\ [9] \end{cases} \Longleftrightarrow 8x \equiv 8 \times 1\ [9] \Longleftrightarrow 8(x - 1) \equiv 0\ [9] \Longleftrightarrow 9 \mid 8(x - 1). \]
Comme 9 est premier avec 8, d'après Gauss, 9 divise $x - 1$. Il existe donc $k \in \Z$ tel que $x - 1 = 9k$. D'où $x = 1 + 9k$. Les solutions de $(E_2)$ sont $\{1 + 9k\,;k \in \Z\}$.

\textbf{c.} En remplaçant $x$ par $1 + 9k$ dans $8x + 9y = -10$, on obtient $y = -2 - 8k$. Les solutions de $(E_1)$ sont $S = \{(1 + 9k\,;-2 - 8k)\,;k \in \Z\}$. Les équations $(E)$ et $(E_1)$ étant équivalentes, $S$ est aussi l'ensemble des solutions de $(E)$.

\textbf{3.} Comme $g$ est la composée d'une symétrie axiale et d'une rotation dont le centre n'appartient pas à l'axe, alors $g$ est une symétrie glissée.

\textbf{4. a.} \underline{Les transformations $f$ et $t_{\vv{BF}} \circ S_{(AC)}$ ne sont pas égales.} Soit $Q$ le centre du rectangle $ABCD$. $f(Q) = S_{(AD)} \circ S_{(AB)} \circ S_{(BD)}(Q) = S_{(AD)} \circ S_{(AB)}(Q) = S_{(AD)}(F) = H$. Mais $t_{\vv{BF}} \circ S_{(AC)}(Q) = t_{\vv{BF}}(Q) = A$. D'où $f \ne t_{\vv{BF}} \circ S_{(AC)}$.

\underline{Montrons plutôt que $S_{(AD)} \circ R = t_{\vv{BF}} \circ S_{(AC)}$.}

\emph{Première solution :}
\begin{align*}
S_{(AD)} \circ R &= S_{(AD)} \circ S_{(BC)} \circ S_{(BG)} = t_{\vv{BE}} \circ S_{(BG)} \\
&= t_{\vv{BF}} \circ t_{\vv{FE}} \circ S_{(BG)} = t_{\vv{BF}} \circ \underbrace{S_{(AC)} \circ S_{(BG)}}_{t_{\vv{FE}} \circ S_{(BG)}} = t_{\vv{BF}} \circ S_{(AC)}
\end{align*}

\emph{Deuxième solution :} $S_{(AD)} \circ R$ est la composée d'une symétrie axiale et d'une rotation dont le centre n'appartient pas à l'axe : c'est une symétrie glissée. Posons $u = S_{(AD)} \circ R$. On a $u(B) = S_{(AD)}(B) = E$ et $u(E) = S_{(AD)}(G) = G$, d'où $u \circ u(B) = G$. On en déduit que le vecteur de $u$ est $\dfrac{1}{2}\vv{BG} = \vv{BF}$. Comme $u(B) = E$, l'axe de $u$ est la droite dirigée par $\vv{BF}$ passant par le milieu de $[BE]$, c'est-à-dire $(AC)$. D'où $S_{(AD)} \circ R = t_{\vv{BF}} \circ S_{(AC)}$.

\textbf{b.} L'axe $(AC)$ étant dirigé par $\vv{BF}$, $S_{(AD)} \circ R_{(B,\frac{\pi}{3})}$ est une symétrie glissée de vecteur $\vv{BF}$ et d'axe $(AC)$.

\exo{3}{}

\textbf{1.} L'équation caractéristique associée à $y'' + 2y' + 2y = 0$ est $r^2 + 2r + 2 = 0$. Elle admet deux racines $r_1 = -1 - i$ et $r_2 = -1 + i$. La solution générale est $y(x) = e^{-x}(c_1\cos x + c_2\sin x)$.

La solution particulière $f$ vérifie $f\!\left(\tfrac{\pi}{2}\right) = e^{-\frac{\pi}{2}}$ et $f'\!\left(\tfrac{\pi}{2}\right) = -e^{-\frac{\pi}{2}}$ :
\[ \begin{cases} f\!\left(\tfrac{\pi}{2}\right) = e^{-\frac{\pi}{2}} \\ f'\!\left(\tfrac{\pi}{2}\right) = -e^{-\frac{\pi}{2}} \end{cases} \Longleftrightarrow \begin{cases} c_2 = 1 \\ c_1 + c_2 = 1 \end{cases} \Longleftrightarrow \begin{cases} c_2 = 1 \\ c_1 = 0 \end{cases} \]
$f$ est la fonction définie sur $\R$ par $f(x) = e^{-x}\sin x$.

\textbf{2. a.} $F$ est une primitive de $f$ si $\forall x \in \R$, $F'(x) = f(x)$. On en déduit que $(-A + B)\cos x + (-A - B)\sin x = \sin x$. Par identification, $-A + B = 0$ et $-A - B = 1$. D'où $A = B = -\dfrac{1}{2}$. $F$ est définie par $F(x) = -\dfrac{e^{-x}(\cos x + \sin x)}{2}$.

\textbf{b.} $\displaystyle\int_{n\pi}^{(n+1)\pi} f(x)\,dx = \left[-\frac{e^{-x}(\cos x + \sin x)}{2}\right]_{n\pi}^{(n+1)\pi} = \frac{e^{-\pi} + 1}{2}(-e^{-\pi})^n$.

\textbf{3. a.} $\forall n \in \N$, $\dfrac{v_{n+1}}{v_n} = \dfrac{\frac{e^{-\pi} + 1}{2}(-e^{-\pi})^{n+1}}{\frac{e^{-\pi} + 1}{2}(-e^{-\pi})^n} = -e^{-\pi}$. $(v_n)$ est une suite géométrique de raison $-e^{-\pi}$ et de premier terme $v_0 = \dfrac{e^{-\pi} + 1}{2}$.

\textbf{b.} $S_n = v_0 + v_1 + \cdots + v_n = \dfrac{e^{-\pi} + 1}{2}\big(1 + (-e^{-\pi}) + \cdots + (-e^{-\pi})^n\big) = \dfrac{1}{2}\big(1 - (-e^{-\pi})^{n+1}\big)$. Comme $|-e^{-\pi}| = e^{-\pi} < 1$, alors $\lim\limits_{n\to+\infty}(-e^{-\pi})^{n+1} = 0$. D'où $\lim\limits_{n\to+\infty} S_n = \dfrac{1}{2}$.

\textbf{2.} On note $(x_i, n_{i\bullet})$ et $(y_j, n_{\bullet j})$ les distributions marginales, avec $\sum_i n_{i\bullet} = \sum_j n_{\bullet j} = N$.
\begin{center}\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|c|}\hline $X$ & 1 & 2 \\\hline $n_{i\bullet}$ & 4 & 6 \\\hline\end{tabular}
\qquad

\[ V(X) = \frac{1}{10}\left[4 \times 1^2 + 6 \times 2^2\right] - 1{,}6^2 = 0{,}24. \quad V(Y) = \frac{1}{10}\left[4(-1)^2 + 4 \times 2^2 + 2 \times 3^2\right] - 1^2 = 2{,}8. \]
\begin{align*}
\mathrm{Cov}(X, Y) &= \frac{1}{10}\big(2(-1) + 1 \times 2 + 1 \times 3 + 2(-2) + 3 \times 4 + 1 \times 6\big) - 1{,}6 \times 1 = 0{,}1.
\end{align*}
D'où le coefficient de corrélation : $\rho_{X,Y} = \dfrac{\mathrm{Cov}(X, Y)}{\sqrt{V(X)}\sqrt{V(Y)}} = \dfrac{0{,}1}{\sqrt{0{,}24} \times \sqrt{2{,}8}} = 0{,}1219$.

\textbf{3.} D'après 1. et sachant que $\overline{X} = 1{,}6$ et $\overline{Y} = 1$, les vecteurs centrés ont pour coordonnées :
\[ X - \overline{X} = (-0{,}6\,;-0{,}6\,;-0{,}6\,;-0{,}6\,;0{,}4\,;0{,}4\,;0{,}4\,;0{,}4\,;0{,}4\,;0{,}4), \]
\[ Y - \overline{Y} = (-2\,;-2\,;1\,;2\,;-2\,;-2\,;1\,;1\,;1\,;2). \]
Le cosinus de l'angle formé par les vecteurs centrés est donné par $\cos\big(\widehat{(X - \overline{X}),(Y - \overline{Y})}\big) = \rho_{X,Y}$. D'où cet angle vaut $\cos^{-1}(0{,}1219) \approx 83^\circ$.
