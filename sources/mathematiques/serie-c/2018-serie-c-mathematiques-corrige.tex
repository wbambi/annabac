\exo{1}{}

\textbf{1.}
\[ (S) : \begin{cases} x \equiv 2\ [36] \\ x \equiv 3\ [25] \end{cases} \Longleftrightarrow \begin{cases} x = 2 + 36a,\ a \in \Z \\ x = 3 + 25b,\ b \in \Z \end{cases} \Longleftrightarrow 2 + 36a = 3 + 25b \Longleftrightarrow (E) : 36a - 25b = 1. \]
On en déduit que le système $(S)$ et l'équation $(E)$ sont équivalents.

\textbf{2.} $36 \times (-9) - 25 \times (-13) = -324 + 325 = 1$. Donc $(-9;-13)$ est solution de $(E) : 36a - 25b = 1$.

\textbf{3.}
\begin{align*}
36a - 25b = 1 &\Longleftrightarrow 36a - 1 = 25b,\ b \in \Z \\
&\Longleftrightarrow 36a - 1 \text{ est un multiple de } 25 \Longleftrightarrow 36a - 1 \equiv 0\ [25] \Longleftrightarrow 36a \equiv 1\ [25]
\end{align*}
On en déduit que $(E)$ et $(E')$ sont équivalentes.

\textbf{4.} Comme $36 \times (-9) = 1 + 25 \times (-13)$, alors $36 \times (-9) \equiv 1\ [25]$. Donc $-9$ est l'inverse de 36 modulo 25.

\textbf{5.}
\[ \begin{cases} 36a \equiv 1\ [25] \\ 36 \times (-9) \equiv 1\ [25] \end{cases} \Longleftrightarrow 36a \equiv 36 \times (-9)\ [25] \Longleftrightarrow 36(a + 9) \equiv 0\ [25] \Longleftrightarrow 25 \mid 36(a + 9). \]
Comme 25 est premier avec 36, d'après Gauss, 25 divise $a + 9$. Il existe donc $k \in \Z$ tel que $a + 9 = 25k$. D'où $a = -9 + 25k$. Les solutions de $(E')$ sont $\{-9 + 25k\,;k \in \Z\}$.

\textbf{6. a.} En remplaçant $a$ par $-9 + 25k$ dans $36a - 25b = 1$, on obtient $b = -13 + 36k$. L'ensemble des solutions de $(E)$ est $\{(-9 + 25k\,;-13 + 36k)\,;k \in \Z\}$.

\textbf{b.} Soit $x$ une solution de $(S)$, de la forme $x = 2 + 36a$ où $a$ vérifie $(E')$. D'après la question précédente, $a = -9 + 25k$ avec $k \in \Z$. D'où $x = 2 + 36(-9 + 25k) = -322 + 900k$.
\[ 0 < x < 50 \Longleftrightarrow 0 < -322 + 900k < 50 \Longleftrightarrow \frac{322}{900} < k < \frac{372}{900}. \]
Il n'existe pas d'entier $k$ compris entre $\dfrac{322}{900}$ et $\dfrac{372}{900}$. Donc le système $(S)$ n'admet pas de solution $x$ telle que $0 < x < 50$.

\textbf{1.} Voir figure.

\textbf{2. a.} $(OF)$ est l'axe focal. De plus, $OC$ est la demi-distance focale. Donc $J$ est le point d'intersection du cercle $(C_1)$ et de la demi-droite $[OF)$.

\textbf{b.} $J$ étant un foyer de $(\Gamma)$, le projeté orthogonal de $J$ sur l'une des diagonales $[AC]$ ou $[BD]$ est un point de la directrice associée à $J$. D'où $(D)$ est la droite perpendiculaire à l'axe focal $(OF)$ passant par ce projeté orthogonal.

\textbf{c.} Notons $J_1$ le second foyer de $(\Gamma)$ ; $(C_d)$ le cercle directeur associé à $J$ (centre $J$, rayon $2 \times OF = 4$) ; $P$ le point d'intersection de $[BJ)$ et de $(C_d)$. $K$ est alors le point d'intersection de la médiatrice de $[J_1 P]$ et de $[BJ)$. En effet,
\[ KJ_1 - KJ = KJ_1 - (KP - JP) = KJ_1 - KP + 4. \]
Comme $K$ est sur la médiatrice de $[J_1 P]$, alors $KJ_1 = KP$. D'où $KJ_1 - KJ = 4$, ce qui prouve que $K$ est un point de l'hyperbole $(\Gamma)$.

\textbf{d.} $[OC)$ est la demi-droite asymptote de $(\Gamma)$ située dans la portion délimitée par les demi-droites $[OF)$ et $[OG)$.

\textbf{e.} Construire $K'$ symétrique de $K$ par rapport à l'axe focal $(OF)$. À partir des points $F$ et $K'$ et sachant que $[OC)$ est une asymptote à $(\Gamma_0)$, on donne une allure de $(\Gamma_0)$.

\textbf{3.} Notons $\Omega$ le centre de la similitude $S$, $h_{(\Omega,2)}$ l'homothétie de centre $\Omega$ et de rapport 2, $S_{(AC)}$ la symétrie axiale d'axe $(AC)$. $S$ s'écrit $S = h_{(\Omega,2)} \circ S_{(AC)}$ avec $\Omega \in (AC)$.

\underline{Centre $\Omega$ :} $S(F) = I$. $S(F) = h_{(\Omega,2)} \circ S_{(AC)}(F) = h_{(\Omega,2)}(G)$. On en déduit que $h_{(\Omega,2)}(G) = I$, soit $2\vv{\Omega G} = \vv{\Omega I}$. Par Chasles, $2\vv{\Omega O} + 2\vv{OG} = \vv{\Omega O} + \vv{OI}$, d'où $\vv{\Omega O} = \vv{OI} - 2\vv{OG} = \vv{0}$ car $I$ est le symétrique de $O$ par rapport à $G$. D'où $\Omega = O$.

\textbf{4. a.} Le rectangle fondamental de $(\Gamma)$ étant un carré, $(\Gamma)$ est une hyperbole équilatère. Comme toute similitude conserve la nature des figures, le rectangle fondamental de $(\Gamma')$ est un carré. Donc $(\Gamma')$ est une hyperbole équilatère.

\textbf{b.} Toute similitude conserve le rapport des distances. On en déduit que $(\Gamma)$ et $(\Gamma')$ ont même excentricité. Donc l'excentricité de $(\Gamma')$ est $\dfrac{OC}{OF} = \sqrt{2}$.

\textbf{c.} $S(O) = O$ et $S(F) = I$. Comme le cercle de centre $O$ et de rayon $OF$ est le cercle principal de $(\Gamma)$, alors le cercle de centre $O$ et de rayon $OI$ est le cercle principal de $(\Gamma')$.

\textbf{d.} $S(A) = h_{(O,2)}(A) \in (AC)$ et $S(C) = h_{(O,2)}(C) \in (AC)$. La droite $(AC)$ est stable par $S$ : c'est une asymptote de $(\Gamma')$. De même, $S(B) = h_{(O,2)}(D) \in (BD)$ et $S(D) = h_{(O,2)}(B) \in (BD)$ : la droite $(BD)$ est stable, c'est la seconde asymptote de $(\Gamma')$.

\textbf{e.} $S(O) = O$ et $S(F) = I$. Comme $(OF)$ est l'axe focal de $(\Gamma)$, alors $(OI)$ est l'axe focal de $(\Gamma')$.

\textbf{2.} $\forall x \in ]0\,;+\infty[$ : $g'(x) > 0$. D'où le tableau de variation :
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccc|}
\hline
$x$ & $0$ & & $+\infty$ \\
\hline
$g'(x)$ & $\|$ & $+$ & \\

\hline
\end{tabular}
\end{center}

\textbf{3. a.} $g\!\left(\tfrac{1}{2}\right) \approx -0{,}17$ et $g(1) \approx 1{,}71$. La fonction $g$ est continue et strictement croissante sur $\left]\tfrac{1}{2}\,;1\right[$. De plus $g\!\left(\tfrac{1}{2}\right) \cdot g(1) < 0$. D'après le théorème des valeurs intermédiaires, il existe un unique réel $\alpha \in \left]\tfrac{1}{2}\,;1\right[$ tel que $g(\alpha) = 0$.

\textbf{b.} $g$ est strictement croissante sur $]0\,;+\infty[$ et $g(\alpha) = 0$. On en déduit que : $\forall x \in ]0\,;\alpha[$, $g(x) < 0$ ; $\forall x \in ]\alpha\,;+\infty[$, $g(x) > 0$.

\partieT{B}
\textbf{4.} $\lim\limits_{x\to 0^+} f(x) = +\infty$ ; $\lim\limits_{x\to+\infty} f(x) = \lim\limits_{x\to+\infty} e^x\left(1 - \dfrac{\ln x}{e^x}\right) = +\infty$.

\textbf{5.} $\forall x \in ]0\,;+\infty[$, $f'(x) = e^x - \dfrac{1}{x} = \dfrac{g(x)}{x}$.

\textbf{6. a.} $f'$ est du signe de $g$ sur $]0\,;+\infty[$.
\begin{center}\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|ccccc|}
\hline
$x$ & $0$ & & $\alpha$ & & $+\infty$ \\
\hline
$f'(x)$ & $\|$ & $-$ & $0$ & $+$ & \\
\hline
$f$ & $+\infty$ & $\searrow$ & $f(\alpha)$ & $\nearrow$ & $+\infty$ \\
\hline
\end{tabular}
\end{center}

\textbf{b.} $f$ est continue sur $]0\,;+\infty[$, strictement décroissante sur $]0\,;\alpha[$ puis croissante sur $]\alpha\,;+\infty[$. Donc $f$ admet un minimum en $\alpha$. $\alpha$ vérifie $f'(\alpha) = \dfrac{\alpha e^\alpha - 1}{\alpha} = 0$, donc $e^\alpha = \dfrac{1}{\alpha}$ et $\alpha = \dfrac{1}{e^\alpha}$. En remplaçant dans $f(\alpha) = e^\alpha - \ln\alpha$ :
\[ f(\alpha) = \frac{1}{\alpha} - \ln\frac{1}{e^\alpha} = \frac{1}{\alpha} + \alpha. \]
Donc le minimum est $f(\alpha) = \alpha + \dfrac{1}{\alpha}$.

\exo{4}{}

On note $(x_i, n_{i\bullet})$ et $(y_j, n_{\bullet j})$ les distributions marginales de $X$ et $Y$, avec $\sum_i n_{i\bullet} = \sum_j n_{\bullet j} = N$.

\textbf{1.}
\begin{center}\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|c|c|}\hline $X$ & 2 & 3 \\\hline $n_{i\bullet}$ & 5 & 8 \\\hline\end{tabular}
\qquad
\begin{tabular}{|c|c|c|c|}\hline $Y$ & $-1$ & 2 & 3 \\\hline $n_{\bullet j}$ & 3 & 3 & 7 \\\hline\end{tabular}
\end{center}

\textbf{2.}
\[ \overline{X} = \frac{1}{N}\sum_{i=1}^2 n_{i\bullet}x_i = \frac{5 \times 2 + 8 \times 3}{13} = \frac{34}{13}, \quad \overline{Y} = \frac{3(-1) + 3 \times 2 + 7 \times 3}{13} = \frac{24}{13}. \]
Les coordonnées du point moyen $G$ sont $\left(\dfrac{34}{13}, \dfrac{24}{13}\right)$.

\textbf{3. a.}
\[ \mathrm{Cov}(X, Y) = \frac{1}{13}\big(2(-2) + 3 \times 6 + 1(-3) + 3 \times 6 + 4 \times 9\big) - \frac{34}{13} \times \frac{24}{13} = \frac{29}{169}. \]

\textbf{b.} L'équation de régression linéaire de $Y$ en $X$ est $Y = aX + b$ où $a = \dfrac{\mathrm{Cov}(X, Y)}{\mathrm{Var}(X)}$ et $b = \overline{Y} - a\overline{X}$.
\[ V(X) = \frac{1}{13}(5 \times 2^2 + 8 \times 3^2) - \left(\frac{34}{13}\right)^2 = \frac{40}{169}. \]
\[ a = \frac{\frac{29}{169}}{\frac{40}{169}} = \frac{29}{40} \quad \text{et} \quad b = \frac{24}{13} - \frac{29}{40} \times \frac{34}{13} = -\frac{1}{20}. \]
D'où l'équation de la droite de régression linéaire : $Y = \dfrac{29}{40}X - \dfrac{1}{20}$.
